/**
 * Clase WebRouterCliStub

 * Adapta una interacción proveniente de línea de comandos a una interacción tipo web service, a efectos de simulación.

 * @package botbasic

{

    {

    }

    {

    }

    {
        global $_post;
        return $_post;
    }
